# Comando para obter tempo de acompanhamento
@bot.command(name="tempoacompanhamento")
async def tempo_acompanhamento(ctx, member: commands.MemberConverter):
    user_id = member.id
    headers = {
        'Client-ID': 'seu_client_id',
        'Authorization': 'Bearer seu_token'
    }
    response = requests.get(f'https://api.twitch.tv/helix/users/follows?from_id={user_id}', headers=headers)
    data = response.json()
    tempo = data['data'][0]['followed_at']
    await ctx.send(f'{member.name} te segue desde: {tempo}')

# Comando para obter pontos do usuário
@bot.command(name="pontos")
async def pontos(ctx, member: commands.MemberConverter):
    user_id = member.id
    headers = {
        'Client-ID': 'seu_client_id',
        'Authorization': 'Bearer seu_token'
    }
    response = requests.get(f'https://api.twitch.tv/helix/channel_points/custom_rewards?broadcaster_id={user_id}', headers=headers)
    data = response.json()
    pontos = data['data'][0]['default_reward']['cost']
    await ctx.send(f'{member.name} tem {pontos} pontos!')
